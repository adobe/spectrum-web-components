import {
  Meta,
  Title,
  Primary,
  Controls,
  Stories,
  ArgTypes,
  Description,
  Subtitle,
  HeaderMdx,
  useOf,
} from '@storybook/addon-docs/blocks';
import {
  ApiTable,
  GettingStarted,
  OverviewStory,
  SpectrumStories,
  StatusBadge,
} from './blocks';

export const checkIsSingleStory = () => {
  const resolvedOf = useOf('meta', ['meta']);
  const { stories } = resolvedOf.csfFile;
  const visibleStories = Object.values(stories).filter(
    (story) => story.tags?.includes('autodocs') || story.tags?.includes('dev')
  );
  return visibleStories.length === 1;
};

export const SingleStoryDescription = () => {
    const isSingleStory = checkIsSingleStory();

    return (isSingleStory ? <Description of="story" /> : null)

}

export const AdvancedExamplesStories = () => {
    const isSingleStory = checkIsSingleStory();

    return (isSingleStory ? null : <Stories includePrimary={false} />)

}

export const slugify = (s) =>
  s
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/(^-|-$)/g, '');

export const ConditionalSection = ({ tag, title, hideTitle = false }) => {
    const resolvedOf = useOf('meta', ['meta']);
    const hasStories = Object.values(resolvedOf.csfFile.stories).some(
        (story) => story.tags?.includes(tag)
    );

    if (!hasStories) {
        return null;
    }

    return (
        <div data-section={tag}>
            {title && <HeaderMdx as="h2" id={slugify(title)}>{title}</HeaderMdx>}
            <SpectrumStories tag={tag} hideTitle={hideTitle} />
        </div>
    );

};
export const ConditionalApi = () => {
const resolvedOf = useOf('meta', ['meta']);
const tags = resolvedOf?.preparedMeta?.tags ?? resolvedOf?.csfFile?.meta?.tags ?? [];

    // Controllers document their API via inline JSDoc tables in description-only
    // stories tagged `api`. They have no custom-element manifest entry, so the
    // CEM-driven ApiTable is not applicable.
    if (tags.includes('controller')) {
        return (
            <>
                <HeaderMdx as="h2" id="api">API</HeaderMdx>
                <SpectrumStories tag="api" hideTitle={true} />
            </>
        );
    }

    return (
        <>
            <HeaderMdx as="h2" id="api">API</HeaderMdx>
            <ApiTable />
        </>
    );

};

export const ConditionalGettingStarted = () => {
    const resolvedOf = useOf('meta', ['meta']);

    const tags = resolvedOf?.csfFile?.meta?.tags ?? [];

    return <GettingStarted tags={tags} />;

};
